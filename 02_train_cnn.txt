import argparse
import csv
import math
import random
import sys
import time
from pathlib import Path

import numpy as np
import torch
from torch import nn
from torch.optim import AdamW
from torch.optim.lr_scheduler import LambdaLR


def register_project_path(project_root: Path):
    project_root = project_root.resolve()
    if str(project_root) not in sys.path:
        sys.path.insert(0, str(project_root))


def fix_random_generators(seed: int):
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)

    torch.backends.cudnn.deterministic = False
    torch.backends.cudnn.benchmark = True


def cosine_warmup_schedule(total_steps: int, warmup_steps: int):
    def schedule(step: int):
        if step < warmup_steps:
            return float(step) / max(1, warmup_steps)

        progress = float(step - warmup_steps) / max(1, total_steps - warmup_steps)
        return 0.5 * (1.0 + math.cos(math.pi * progress))

    return schedule


def batch_frame_f1(logits: torch.Tensor, targets: torch.Tensor, activation_threshold: float):
    with torch.no_grad():
        probs = torch.sigmoid(logits)
        predicted = probs >= activation_threshold

        reference = targets.bool()
        predicted = predicted.bool()

        tp = (reference & predicted).sum().item()
        fp = (~reference & predicted).sum().item()
        fn = (reference & ~predicted).sum().item()

        precision = tp / (tp + fp + 1e-8)
        recall = tp / (tp + fn + 1e-8)
        f1 = 2 * precision * recall / (precision + recall + 1e-8)

        return f1


def train_epoch(model, loader, criterion, optimizer, scheduler, device, activation_threshold, use_amp):
    model.train()

    total_loss = 0.0
    total_f1 = 0.0
    steps = 0

    scaler = torch.cuda.amp.GradScaler(enabled=use_amp)

    for spectrogram_batch, piano_roll_batch in loader:
        spectrogram_batch = spectrogram_batch.to(device, non_blocking=True)
        piano_roll_batch = piano_roll_batch.to(device, non_blocking=True)

        optimizer.zero_grad(set_to_none=True)

        with torch.cuda.amp.autocast(enabled=use_amp):
            logits = model(spectrogram_batch)
            loss = criterion(
                logits.reshape(-1, logits.shape[-1]),
                piano_roll_batch.reshape(-1, piano_roll_batch.shape[-1]),
            )

        scaler.scale(loss).backward()
        scaler.step(optimizer)
        scaler.update()

        scheduler.step()

        total_loss += loss.item()
        total_f1 += batch_frame_f1(logits.detach(), piano_roll_batch.detach(), activation_threshold)
        steps += 1

    return total_loss / max(1, steps), total_f1 / max(1, steps)


@torch.no_grad()
def validate_epoch(model, loader, criterion, device, activation_threshold):
    model.eval()

    total_loss = 0.0
    total_f1 = 0.0
    steps = 0

    for spectrogram_batch, piano_roll_batch in loader:
        spectrogram_batch = spectrogram_batch.to(device, non_blocking=True)
        piano_roll_batch = piano_roll_batch.to(device, non_blocking=True)

        logits = model(spectrogram_batch)

        loss = criterion(
            logits.reshape(-1, logits.shape[-1]),
            piano_roll_batch.reshape(-1, piano_roll_batch.shape[-1]),
        )

        total_loss += loss.item()
        total_f1 += batch_frame_f1(logits, piano_roll_batch, activation_threshold)
        steps += 1

    return total_loss / max(1, steps), total_f1 / max(1, steps)


def main():
    parser = argparse.ArgumentParser()

    parser.add_argument("--project-root", type=str, required=True)
    parser.add_argument("--features-root", type=str, required=True)

    parser.add_argument("--representation", type=str, default="mel", choices=["mel", "cqt"])

    parser.add_argument("--epochs", type=int, default=120)
    parser.add_argument("--batch-size", type=int, default=16)
    parser.add_argument("--segment-frames", type=int, default=1024)
    parser.add_argument("--loader-workers", type=int, default=4)

    parser.add_argument("--initial-learning-rate", type=float, default=7e-4)
    parser.add_argument("--regularization-weight", type=float, default=1e-4)
    parser.add_argument("--warmup-fraction", type=float, default=0.05)

    parser.add_argument("--active-note-weight", type=float, default=4.0)
    parser.add_argument("--activation-threshold", type=float, default=0.55)

    parser.add_argument("--seed", type=int, default=42)
    parser.add_argument("--enable-mixed-precision", action="store_true")

    args = parser.parse_args()

    project_root = Path(args.project_root)
    features_root = Path(args.features_root)

    register_project_path(project_root)
    fix_random_generators(args.seed)

    from models.cnn_amt import CNNTemporal
    from scripts.dataset_helpers import create_dataloaders

    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    use_amp = device.type == "cuda" and args.enable_mixed_precision

    frequency_bins = 229 if args.representation == "mel" else 252
    pitch_count = 88

    checkpoint_dir = project_root / "checkpoints"
    checkpoint_dir.mkdir(parents=True, exist_ok=True)

    checkpoint_path = checkpoint_dir / f"cnn_{args.representation}_best.pt"
    history_path = checkpoint_dir / f"cnn_{args.representation}_history.csv"

    print("Project root:", project_root)
    print("Features root:", features_root)
    print("Representation:", args.representation)
    print("Frequency bins:", frequency_bins)
    print("Device:", device)
    print("Mixed precision:", use_amp)
    print("Checkpoint:", checkpoint_path)

    train_loader, val_loader = create_dataloaders(
        root_dir=str(features_root),
        spec_type=args.representation,
        batch_size=args.batch_size,
        chunk_len=args.segment_frames,
        num_workers=args.loader_workers,
    )

    print("Train batches:", len(train_loader))
    print("Validation batches:", len(val_loader))

    model = CNNTemporal(
        n_freq_bins=frequency_bins,
        n_pitches=pitch_count,
    ).to(device)

    parameter_count = sum(p.numel() for p in model.parameters())
    trainable_parameter_count = sum(p.numel() for p in model.parameters() if p.requires_grad)

    print("Total parameters:", f"{parameter_count:,}")
    print("Trainable parameters:", f"{trainable_parameter_count:,}")

    pos_weight = torch.full(
        size=(pitch_count,),
        fill_value=args.active_note_weight,
        device=device,
    )

    criterion = nn.BCEWithLogitsLoss(pos_weight=pos_weight)

    optimizer = AdamW(
        model.parameters(),
        lr=args.initial_learning_rate,
        weight_decay=args.regularization_weight,
    )

    total_steps = args.epochs * len(train_loader)
    warmup_steps = int(args.warmup_fraction * total_steps)

    scheduler = LambdaLR(
        optimizer,
        lr_lambda=cosine_warmup_schedule(total_steps, warmup_steps),
    )

    print("Total training steps:", total_steps)
    print("Warmup steps:", warmup_steps)

    best_val_f1 = 0.0
    start_time = time.time()

    with open(history_path, "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(
            f,
            fieldnames=[
                "epoch",
                "train_loss",
                "train_frame_f1",
                "val_loss",
                "val_frame_f1",
                "learning_rate",
                "epoch_time_sec",
            ],
        )
        writer.writeheader()

        for epoch in range(1, args.epochs + 1):
            epoch_start = time.time()

            train_loss, train_f1 = train_epoch(
                model=model,
                loader=train_loader,
                criterion=criterion,
                optimizer=optimizer,
                scheduler=scheduler,
                device=device,
                activation_threshold=args.activation_threshold,
                use_amp=use_amp,
            )

            val_loss, val_f1 = validate_epoch(
                model=model,
                loader=val_loader,
                criterion=criterion,
                device=device,
                activation_threshold=args.activation_threshold,
            )

            epoch_time = time.time() - epoch_start
            lr = optimizer.param_groups[0]["lr"]

            writer.writerow(
                {
                    "epoch": epoch,
                    "train_loss": train_loss,
                    "train_frame_f1": train_f1,
                    "val_loss": val_loss,
                    "val_frame_f1": val_f1,
                    "learning_rate": lr,
                    "epoch_time_sec": epoch_time,
                }
            )
            f.flush()

            print(
                f"[Epoch {epoch:03d}/{args.epochs}] "
                f"train_loss={train_loss:.5f}, train_F1={train_f1:.5f} | "
                f"val_loss={val_loss:.5f}, val_F1={val_f1:.5f} | "
                f"lr={lr:.8f} | "
                f"time={epoch_time:.1f}s"
            )

            if val_f1 > best_val_f1:
                best_val_f1 = val_f1

                torch.save(
                    {
                        "model_state_dict": model.state_dict(),
                        "representation": args.representation,
                        "frequency_bins": frequency_bins,
                        "pitch_count": pitch_count,
                        "best_val_frame_f1": best_val_f1,
                        "run_config": vars(args),
                    },
                    checkpoint_path,
                )

                print(f"New best validation Frame F1={best_val_f1:.5f}. Saved.")

    elapsed = time.time() - start_time

    print("\nTraining finished.")
    print("Best validation Frame F1:", f"{best_val_f1:.5f}")
    print("Total training time, hours:", f"{elapsed / 3600:.2f}")
    print("History:", history_path)
    print("Best checkpoint:", checkpoint_path)


if __name__ == "__main__":
    main()
